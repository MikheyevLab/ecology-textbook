4. Modeling a single population¶

r <- 1
s <- 0
dt <- 1
t <- 0
N <- 3
timeSteps <- 14
population <- data.frame(time = integer(), exponential = integer())

for(i in 1:timeSteps) {
  population[i, ] = c(t, N)
  dN <- (r + s * N) * N * dt
  N <- N + dN
  t <- t + dt
  
}

{
  population[i, "orthologistic"] <- N

  N <- N + dN
